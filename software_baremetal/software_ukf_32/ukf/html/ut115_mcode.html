1function [y,Y,P,Y1]=ut(eq,X,Wm,Wc,n,R)
 2%Unscented Transformation
 3%Input:
 4%        f: nonlinear map
 5%        X: sigma points
 6%       Wm: weights for mean
 7%       Wc: weights for covraiance
 8%        n: numer of outputs of f
 9%        R: additive covariance
10%Output:
11%        y: transformed mean
12%        Y: transformed smapling points
13%        P: transformed covariance
14%       Y1: transformed deviations
15
16L=size(X,2);
17y=zeros(n,1);
18Y=zeros(n,L);
19for k=1:L
20    if eq==0 
21        Y(:,k)=nonlinear_state_eq(X(:,k));
22    else
23        Y(:,k)=measurement_eq(X(:,k));
24    end     
25    y=y+Wm(k)*Y(:,k);       
26end
27Y1=Y-y(:,ones(1,L));
28P=Y1*diag(Wc)*Y1'+R;

1function A=Cholesky(A)
 2% Cholesky Factorization for symmetric positive definite matrix
 3% Algorithm 2.7 Heath, p.86
 4% Factorize A such that A = L*L',
 5% where L is a lower triangular matrix whose diagonal entries are not
 6% necessarily unity
 7% In the output the lower triangular part of A is over-written by L
 8% The upper triangular part of A is left unchanged and 
 9% has no significance
10% K. Ming Leung, 01/25/03
11
12[n nn]=size(A);
13for k=1:n
14    A(k,k)=sqrt(A(k,k));
15    A(k+1:n,k)=A(k+1:n,k)/A(k,k);
16    for j=k+1:n
17        A(j:n,j)=A(j:n,j)-A(j,k)*A(j:n,k);
18    end
19end
